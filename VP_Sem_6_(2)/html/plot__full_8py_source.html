4"""!
5@file plot_full.py
6@brief Visualization of the numerical solution and comparison with the exact one.
7@details The script reads a binary file solution.bin containing the grid and numerical solution values,
8 creates 3D plots of the exact solution, numerical solution, and absolute error, as well as a 2D error map.
9 Uses matplotlib and numpy.
10"""

12import numpy as np

18def u_exact(x: Union[float, np.ndarray], y: Union[float, np.ndarray]) -> np.ndarray:
19 """!
20 @brief Exact analytical solution of the problem.
21 @param x x-coordinate (scalar or numpy array).
22 @param y y-coordinate (scalar or numpy array).
23 @return u(x,y) = x^2 + cos^2(y*x).
24 """
25 return x**2 + np.cos(y * x)**2

27def read_full_solution(filename: str) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
28 """!
29 @brief Reads a binary solution file created by OutputHandler::save_full_solution.
30 @param filename File name (e.g., 'solution.bin').
31 @return Tuple (X, Y, u_num), where X, Y are coordinate matrices (meshgrid),
32 u_num is a 2D array of the numerical solution.
33 @throws ValueError if the file is too short or has an invalid format.
34 @details File format:
35 - first 8 bytes: number of nodes along x (nx, size_t)
36 - next 8 bytes: number of nodes along y (ny, size_t)
37 - then nx*ny double values in row-major order (j=0..ny-1, i=0..nx-1)
38 """
39 with open(filename, 'rb') as f:
40 nx_bytes = f.read(8)
41 ny_bytes = f.read(8)
42 if len(nx_bytes) < 8 or len(ny_bytes) < 8:
43 raise ValueError("File too short: cannot read dimensions")
44 nx = struct.unpack('Q', nx_bytes)[0]
45 ny = struct.unpack('Q', ny_bytes)[0]
46
47 total = nx * ny
48 data_bytes = f.read(total * 8)
49 if len(data_bytes) < total * 8:
50 raise ValueError("File too short: cannot read all data")
51
52 data = struct.unpack('d' * total, data_bytes)
53
54 u_num = np.array(data).reshape((ny, nx))
55 x = np.linspace(0, 1, nx)
56 y = np.linspace(0, 1, ny)
57 X, Y = np.meshgrid(x, y)
58 return X, Y, u_num
